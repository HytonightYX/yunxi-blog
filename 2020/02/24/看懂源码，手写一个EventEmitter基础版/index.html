手写一个EventEmitter基础版

Posted on 2020-02-24 Views: Valine:

说到前端的设计模式，不得不提到观察者模式以及它的变体发布/订阅模式。比如常见的 Webpack 插件机制，Mobx 等等，当然堪称经典的还是 Node.js 中的 Event。

手写

有几个注意点：

  • call() 和 apply() 是有性能差异的，老版本的 Event 中，少于 3 个参数的使用 call()，否则用 apply()；不过现在都用 Reflect 重写了；
  • 这里用了 Map() 来存储 <事件，回调> 的键值对；
  • 一个没想出来的问题：为什么回调数组只有一个元素的时候要变为单个 function，多个回调的时候放数组，不能单个回调的时候放数组吗，比如 [fn] ？
/**
 * 手写 Event
 * Event 源码地址 : 
 * https://github.com/Gozala/events/blob/master/events.js
 */
class EventEmitter {
  constructor() {
    // 存储 事件/回调 键值对
    this._events = this._events || new Map()
    // 监听上限
    this._maxListeners = this._maxListeners || 10
  }
}

/**
 * 在少数参数时call的性能更好,多个参数时apply性能更好 
 * 不过现相应的call/apply操作用Reflect新关键字重写了
 */
EventEmitter.prototype.emit = function (eventName, ...args) {
  let handler
  // 从存储事件键值对中获取对应事件的回调函数
  handler = this._events.get(eventName)
  console.log(handler)

  if (Array.isArray(handler)) {
    for (let i = 0; i < handler.length; i++) {
      if (args.length > 0) {
        handler[i].apply(this, args)
      } else {
        handler[i].call(this)
      }
    }
  }

  return true
}

/**
 * 监听名为 name 的事件
 */
EventEmitter.prototype.addListener = function (eventName, fn) {
  // 获取对应事件名称的函数清单
  const handler = this._events.get(eventName)
  // 将 eventName 事件以及对应的 fn 函数放入 this._events 中存储
  if (!handler) {
    // 如果没有这个键值对，新建它
    this._events.set(eventName, [fn])
  } else {
    // 如果已有这个键，则将 fn 放入对应的数组
    handler.push(fn)
  }
}

/**
 * 移除监听器
 */
EventEmitter.prototype.removeListener = function (eventName, fn) {
  if (this._events === undefined) return this
  // handler 是一个 Array
  const handler = this._events.get(eventName)

  if (handler) {
    let position = -1
    for (let i = 0; i < handler.length; i++) {
      if (handler[i] === fn) {
        position = i
      }
    }

    if (position > -1) {
      _spliceOne(handler, position)
    }
  }

  return this
}

/**
 * 工具函数，从 list 中删除下标为 index 的元素 
 */
function _spliceOne(list, index) {
  for (; index + 1 < list.length; index++)
    list[index] = list[index + 1]
  list.pop()
}

const emitter = new EventEmitter()

function doFE(man) {
  console.log(`写前端 ${man}`)
}

function doBE(man) {
  console.log(`写后端 ${man}`)
}

function doTest(man) {
  console.log(`做测试 ${man}`)
}

// 添加三个监听器
emitter.addListener('work', doFE)
emitter.addListener('work', doBE)
emitter.addListener('work', doTest)
emitter.emit('work', 'Asuna')

// 移除一个监听器
emitter.removeListener('work', doFE)
emitter.emit('work', 'Asuna')
