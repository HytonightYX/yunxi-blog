数据库原理实验

基本操作

创建数据库
create database DBLab;

使用数据库
use DBLab;

创建表
DROP TABLE IF EXISTS `student`;
CREATE TABLE `student` (
  `ID` varchar(5) NOT NULL,
  `name` varchar(20) NOT NULL,
  `dept_name` varchar(20) DEFAULT NULL,
  `tot_cred` decimal(3,0) DEFAULT NULL,
  `birthday` date DEFAULT NULL,
  PRIMARY KEY (`ID`),
  KEY `dept_name` (`dept_name`),
  CONSTRAINT `student_ibfk_1` FOREIGN KEY (`dept_name`) REFERENCES `department` (`dept_name`) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

删除数据表
drop table student;

删除数据库
drop database DBLab;

查询

1.查询所有计算机系课程
select * 
from course 
where dept_name='Comp. Sci';

2.列出所有计算机系绩点在50以上的学生姓名和绩点
select name, tot_cred 
from student 
where dept_name='Comp. Sci' and tot_cred >= 50;

3.列出所有介绍性课程(以Intro开头,结尾,包含)
select * from course where title like 'Intro%';
select * from course where title like '%Intro';
select * from course where title like '%Intro%';

4.包含正好6个字符的学生名字
select ID, name, dept_name 
from student 
where name like '______';

    以r为第2个字符的6个字符的学生名字
select ID, name, dept_name 
from student 
where name like '_r____';

5.选了CS-101没选CS-190的学生学号
select distinct ID 
from takes 
where course_id='CS-101' and ID not in 
    (select ID from takes where course_id='CS-190');

6.查询所有学生每一门课的成绩并且按照学号排序
select ID, course_id, grade from takes order by ID;

7.查询所有学生每一门课成绩,打印学号,学生姓名,课程号和成绩,按照课程名/学号(逆序)排序
select takes.ID, student.name, takes.course_id, takes.grade 
from takes, student 
where takes.ID = student.ID;

8.查询所有学生每一门课成绩,打印学号,学生姓名,课程号和成绩,按照课程名/学号(逆序)排序
select takes.ID, student.name, course.title, takes.grade 
from takes, student, course 
where takes.ID = student.ID and takes.course_id = course.course_id 
order by course.title, takes.ID desc;

考试题

3.查询不及格人次数最多的系
select dept_name
from ( takes as T
    left join student as S
    on T.ID = S.ID)
where (grade = 'A'or grade = 'A-')
group by dept_name
order by count(*) desc
limit 1;

4.查询所有介绍性课程（课程名以Intro开头）
select title from course where title like 'Intro%';

5.查询所有学生每一门课程的成绩，打印出学号、学生姓名、课程名称和成绩，并首先按课程名排序，再按学号排序
select S.ID, S.name, C.title, T.grade
from student as S, course as C, takes as T
where S.ID = T.ID and C.course_id = T.course_id
order by C.title, S.ID;

6.创建一个视图，内容包含所有课程成绩合格（成绩不为F或空）的学生姓名、课程名称、成绩
create view v as 
select S.name, C.title, T.grade
from student as S, takes as T, course as C
where S.ID = T.ID
    and (T.grade > 'F' or T.grade < 'F')
    and (T.course_id = C.course_id);

8.按以下规则修改学生tot_cred:如果学生tot_cred<100,则加5分，但是增加后的分数不能超过100分
update student set tot_cred=
case
when tot_cred+5<100 then tot_cred+5
when tot_cred<100 and tot_cred+5>=100 then 100
when tot_cred>=100 then tot_cred
end;

9.查询所有上了“计算机科学导论”（Intro. to Comp. Sci.）课程的学生的平均绩点(tot_cred)
select avg(S.tot_cred)
from student as S, course as C, takes as T
where S.ID = T.ID
      and T.course_id = C.course_id
      and C.title = 'Intro. to Computer Science';

10.查询每个系的平均成绩，只打印出平均成绩大于80的系的名称和平均成绩，并按平均成绩从大到小排序
select dept_name,avg(tot_cred)
from student
group by dept_name
having avg(tot_cred)>80
order by avg(tot_cred) desc;

有些课程会被不同系的学生学习，查询上课学生所属的系个数最多的课程。
select title
from course
where course_id = (
  select course_id
  from takes as T, student as S
  where T.ID = S.ID 
  group by course_id
  order by count(distinct dept_name) desc
  limit 1
);

查询“Introduction to Comp. Sci.”课程每个成绩的学生数
select grade as '成绩', count(*) as '人数'
from takes as T, course as C
where T.course_id = C.course_id
  and C.title = 'Intro. to Computer Science'
group by grade;
